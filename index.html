        <button id="fetch-data-btn" class="px-6 py-3 bg-blue-500 text-white font-medium rounded-xl hover:bg-blue-600 transition-all duration-200">

            Fetch Employees

        </button>

        <div id="employee-list" class="mt-6 text-left"></div>

    </div>


    <script>

        document.getElementById('fetch-data-btn').addEventListener('click', async () => {

            const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbxUw8hHVGqmDT304rpY_LvFWBWIlDQWjutAt1tCcMxz8hvGKokTbS1m3qGkODUF02aXzQ/exec';

            const employeeListDiv = document.getElementById('employee-list');


            employeeListDiv.innerHTML = '<p class="text-gray-500">Fetching data...</p>';


            try {

                const response = await fetch(SCRIPT_URL, {

                    method: 'GET',

                    mode: 'cors',

                });

                const data = await response.json();

                

                if (data.employees && data.employees.length > 1) {

                    const employees = data.employees.slice(1);

                    const listHtml = employees.map(employee => `

                        <div class="p-3 my-2 bg-gray-100 rounded-lg shadow-sm">

                            <p class="font-semibold">${employee.Name}</p>

                            <p class="text-sm text-gray-600">ID: ${employee.ID} | Salary: $${employee['Monthly Salary']}</p>

                        </div>

                    `).join('');

                    employeeListDiv.innerHTML = listHtml;

                } else {

                    employeeListDiv.innerHTML = '<p class="text-gray-500">No employees found in the spreadsheet.</p>';

                }

            } catch (error) {

                console.error('Error fetching data:', error);

                employeeListDiv.innerHTML = '<p class="text-red-500">Error fetching data. Please check the script URL and permissions.</p>';

            }

        });

    </script>
